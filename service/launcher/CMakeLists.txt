# 查找当前目录下的所有源文件
# 并将名称保存到 DIR_SRCS 变量
aux_source_directory(. LIST_SRCS)

include_directories(../../engine/inc)
include_directories(../../3rd/tinyxml2/include)

# 链接库
target_link_libraries(launcher ${LINK_LIBS})
